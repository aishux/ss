WITH ranked_data AS (
    SELECT *,
           ROW_NUMBER() OVER (PARTITION BY ID ORDER BY REPORTING_DATE DESC) AS rn
    FROM your_table
)
SELECT 
    COST_CENTER, 
    CLIENT_ADVISOR, 
    CONCAT(CLIENT_ADVISOR_F_NAME, ' ', CLIENT_ADVISOR_L_NAME) AS CLIENT_ADVISOR_NAME, 
    ID AS PARTNER_ID
FROM ranked_data
WHERE rn = 1;
